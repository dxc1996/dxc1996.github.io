laravel门卫

2月 24, 2017 发布在 后端

配置cofig/auth.php

'guards' => [
        'web' => [
            'driver' => 'session',
            'provider' => 'users',
        ],

        'api' => [
            'driver' => 'token',
            'provider' => 'users',
        ],
    ],

    'providers' => [
        'users' => [
            'driver' => 'eloquent',
            'model' => App\User::class,
        ],

        // 'users' => [
        //     'driver' => 'database',
        //     'table' => 'users',
        // ],
    ],

当有一个新模块的admin 现在guards数组加一个数组

'admin' => [
            'driver' => 'session',
            'provider' => 'admin',
 ],

在providers数组加

'admin_users' => [
            'driver' => 'eloquent',
            'model' => App\Admin\Model\User::class,
        ],

然后再app\http\controller\auth\logincontroller中重写guard()
方法

找详细的方法要在use的traits中找。
